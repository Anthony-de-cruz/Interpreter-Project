# Pseudo-Carrier waves.
func carrier = x^2 - x^4;
#plot carrier;

# Positive envelop to modulate carrier
func envelope = (x^2 + 2) / (x^2 + 10);
#plot envelope;

let phase = 0;
while phase < 4000 {
    let phase = phase + 1000;

    # Phase-shift the carrier (to get an variable shift)
    func shiftedCarrier = carrier + phase;

    # Plot the carrier, modulating with an envelope.
    plot shiftedCarrier * envelope;
}
