# Carrier waves
func carrierA = x^2 - 4*x;
func carrierB = -x^3 / 6 + x;

# Positive envelop to modulate carrier
func envelope = (x^2 + 10) / (x^2 + 51);

let phase = 0;

while phase < 200 {
    let phase = phase + 25;

    # Phase-shifted carriers (to get an variable shift)
    let waveA = carrierA + phase;
    let waveB = carrierB - phase / 2;

    # Plot the combined carriers, modulating with an envelope.
    plot (waveA - waveB) * envelope;
}
